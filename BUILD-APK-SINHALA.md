    steps:
    - uses: actions/checkout@v3
    
    - name: Setup Node.js
      uses: actions/setup-node@v3

        node-version: 18
        
    - name: Setup Java JDK

    - name: Build APK

        chmod +x gradlew

    - name: Upload APK
